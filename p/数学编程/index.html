from sympy import *
def grad_ascent(x0,f1x,x):
    epsilon = 1e-6
    step_size =1e-4
    x_old = x0
    x_new = x_old + step_size*f1x.subs(x,x_old).evalf()
    while abs(x_old - x_new) > epsilon:
        x_old = x_new
        x_new = x_old + step_size*f1x.subs(x,x_old).evalf()
        return x_new
